Quick Start

Install & predict in minutes

R (MINTer + estiMINT)

if (!requireNamespace("devtools", quietly = TRUE)) install.packages("devtools")
devtools::install_github("CosmoNaught/MINTer")
devtools::install_github("CosmoNaught/estiMINT")  # EIR + cases
# Optional DB helper
devtools::install_github("CosmoNaught/segMINT")

library(MINTer)
results <- run_mint_scenarios(
  prev_vec = c(0.2,0.4,0.6), Q0_vec = c(0.65,0.75,0.85),
  phi_vec = c(0.45,0.65,0.75), irs_future_vec = c(0.2,0.4,0.5)
)
create_scenario_plots(results$prevalence)

Python (MINTe)

pip install -e .  # from the cloned repo (when published: pip install minte)

from minte import MalariaForecast
f = MalariaForecast(db_path="data.duckdb", predictor="cases", window_size=30)
results = f.run(param_limit=1000, sim_limit=8, output_dir="results")
print(results["gru"]["test_metrics"]["r2"])  # R^2 for GRU
Ecosystem

The MINTverse packages

MINTer (R)

Main orchestration layer: direct emulation or full pipeline with malariasimulation ‚Üí DuckDB ‚Üí emulation. Handles plotting & scenario design.

View on GitHub See demo

estiMINT (R)

Gradient‚Äëboosted trees & random forests to convert prevalence & coverage into initial EIR and project annual cases per 1000 (years 2‚Äì5).

View on GitHub See metrics

MINTe (Python)

Efficient RNNs (GRU/LSTM) for prevalence time‚Äëseries. CLI and Python API, optional CUDA, and built‚Äëin hyperparameter tuning.

View on GitHub Try CLI

Copy‚Äëpaste examples

Direct emulation (R)

library(MINTer)
scenarios <- create_scenarios(
  eir = c(5.2, 35.8, 180.5),
  itn_use = c(0.25, 0.55, 0.85),
  irs_use = c(0.10, 0.35, 0.70),
  lsm = c(0.05, 0.45, 0.85)
)
results <- run_malaria_emulator(scenarios, predictor = "prevalence", model_types = c("LSTM","GRU"))
create_scenario_plots(results, output_dir = "output/plots")

Full pipeline (R + DuckDB)

library(MINTer); library(segMINT)
# Run malariasimulation externally, then ingest:
segMINT::create_database(dir = "Data/Database", file_name = "malariasim.duckdb",
  table_name = "sim_results", data_dir = "Data/Simout")
# Counter‚Äëfactuals on scenario 3
cases <- run_malaria_emulator(db_path = "Data/Database/malariasim.duckdb",
  param_index = 3, predictor = "cases", counterfactual = list(eir = c(0.5, 60, 300)))

Python CLI (MINTe)

minte \
  --db-path data.duckdb \
  --table-name simulation_results \
  --predictor cases \
  --use-cyclical-time \
  --window-size 30 \
  --param-limit 2048 \
  --sim-limit 8 \
  --device cuda \
  --output-dir results

Prevalence ‚Üí EIR (estiMINT)

library(estiMINT)
field <- data.frame(
  prevalence = c(0.20, 0.40), Q0 = c(0.65, 0.75), phi_bednets = c(0.45, 0.65),
  itn_use = c(0.4, 0.6), irs_use = c(0.2, 0.4), seasonal = c(0,1), routine = c(0,0)
)
pretrained <- load_pretrained_eir_models()
xgb <- if (is.list(pretrained$xgboost) & & "model" %in% names(pretrained$xgboost)) pretrained$xgboost$model else pretrained$xgboost
pred <- predict_initial_eir(xgb, field, pretrained$feature_cols)
print(round(pred, 2))
